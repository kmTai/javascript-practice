Higher-order Function

1. Pass functions to functions

            
                const normalFn = function () {
                    console.log("@normalFn");
                };
                
                function higherFn(arg) {
                    console.log("@higherFn");
                    arg();
                }
                
                //higherFn takes normalFn as an argument
                higherFn(normalFn);

2. Return functions from functions

            
                function higherFnWithReturn() {
                    return normalFn;
                }
                
                //return a function 
                console.log(higherFnWithReturn());
                //execute the return function with extra()  
                higherFnWithReturn()();

3. Create functions within functions

            
                function fun1(a) {
                  function fun2(b) {
                    return a + b;
                  }
                  return fun2;
                }
              
                console.log(fun1(1)(2)); //3

P.S.
Callback is a form of higher-order function.
